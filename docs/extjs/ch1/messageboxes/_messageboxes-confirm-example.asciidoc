==== Example

[source, javascript]
----
Ext.Msg.confirm('Are you sure?', 
   'Do you want to delete this widget?', function(btn, text){
   if(btn === 'yes'){
      Ext.Msg.alert('Deleted', 'The widget has been deleted');
   }
   else{
      Ext.Msg.alert('Not Deleted', 'The widget was not deleted');
   }
});
