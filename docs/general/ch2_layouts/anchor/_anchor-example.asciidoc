==== Example

[source, javascript]
----
Ext.create('Ext.form.Panel', {
    title: 'Form Panel with anchor layout',
    bodyPadding: 5,
    width: 350,

    layout: 'anchor',
    defaults: {
        anchor: '100%'
    },

    defaultType: 'textfield',
    items: [{
        fieldLabel: 'First Name',
        name: 'first',
        allowBlank: false,
        anchor: '70%'
    },{
        fieldLabel: 'Last Name',
        name: 'last',
        allowBlank: false
    },
    {
        fieldLabel: 'Age',
        name: 'age',
        allowBlank: false,
        anchor: '40%'
    }],
    renderTo: Ext.getBody()
});
----
