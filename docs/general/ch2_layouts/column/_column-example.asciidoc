==== Example

[source, javascript]
----
Ext.create('Ext.form.Panel', {
    title: 'Column Layout',
    bodyPadding: 8,
    labelWidth: 80,
    width: 550,
    layout:'column',
    defaults:{
        xtype: 'textfield',
        padding: 3,
        columnWidth: 0.33,   //change this to 0.33      
    },
    items: [{
        fieldLabel: 'First Name',     
    },{
        fieldLabel: 'Middle Name',
    },{
        fieldLabel: 'Last Name',
    },{
        fieldLabel: 'Nick Name',
    }, {
        fieldLabel: 'Age'
    },
    {
        fieldLabel: 'Phone'
    }],
    renderTo: Ext.getBody()
});
----
